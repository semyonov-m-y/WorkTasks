CREATE TABLE IF NOT EXISTS T_TESTS (
test_id  INT  NOT NULL AUTO_INCREMENT PRIMARY KEY,
test_name VARCHAR(200)  NOT NULL
);

CREATE TABLE IF NOT EXISTS T_DUMP (
dump_id  INT  NOT NULL AUTO_INCREMENT PRIMARY KEY,
dump_name VARCHAR(200)  NOT NULL
);

CREATE TABLE IF NOT EXISTS T_TEST_DUMP (
test_id  INT  NOT NULL AUTO_INCREMENT PRIMARY KEY,
dump_id VARCHAR(200)  NOT NULL
FOREIGN KEY (test_id) REFERENCES T_TESTS (test_id),
FOREIGN KEY (dump_id) REFERENCES T_DUMP (dump_id)
);
